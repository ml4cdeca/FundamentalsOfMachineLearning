import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

from scipy.optimize import least_squares

For each point on a circle, we know that $$(x-c_x)^2 + (y-c_y)^2 = r^2.$$ Using three points, we begin by shifting the coordinate system so that the first point is in the origin. The equation for this point is then simply $$c_x^2 + c_y^2 = r^2.$$ For the other two points, explicitly sqaring the parantheses and subsituting $r^2$, we find that the equations for the two remaining points simplify to $$2x_i c_x + 2y_i c_y = x_i^2 + y_i^2,$$ with $i \in \{2,3\}$, as the $c_x^2$-terms cancel out.

This is now simply a linear system of 2 equations in $c=(c_x, c_y)$. By putting the coefficients of both equations in matrix/vector form, we get a problem of the type $$ A c = b, $$ which can be easily solved.

def cirle_from_points(points):

    # Moving the set of points to the orgin simplifies the problem
    offset = points[0]
    p = points - offset

    # 2x2 linear eq:
    A = 2 * p[1:, :]
    b = p[1:, 0]**2 + p[1:, 1]**2

    try:
        c = np.dot(np.linalg.inv(A), b)
        r = np.sqrt(c[0]**2 + c[1]**2)
    except np.linalg.linalg.LinAlgError:
        # If the points are on a straight line, there is no solution
        return np.NaN * np.zeros(2), np.NaN
    else:
        return c + offset, r

data = np.load('circles.npy')

eps = 0.07              # Epsilon, the inlier tolerance
N_ransac = int(1e4)     # Number of RANSAC iterations
N_cicles = 6            # Number of cirlces to find

# Perform one ransac iteration:
def ransac_iter():
    # 1. Choose 3 points
    points = data[np.random.choice(all_indexes, 3, replace=False)]
    # 2. Calculate the circle through the points
    c, r = cirle_from_points(points)
    # 3. Check how far all other points are from the circle
    dist_from_circ = np.abs((data[:,0] - c[0])**2 
                          + (data[:,1] - c[1])**2
                          - r**2)
    # 4. All below the epsilon theshhold are inliers
    inliers = dist_from_circ < eps**2

    return inliers, c, r


ransac_results = []
for n in range(N_cicles):

    all_indexes = np.arange(data.shape[0], dtype=int)   # Re-index the remaining data
    inliers_best = np.zeros(data.shape[0], dtype=bool)  # Inliers of the best solution
    score_best = 0                                      # Score of the best solution
    c_best, r_best = np.zeros(2), 0.                    # Parameters of the best solution

    for i in range(N_ransac):
        # From a single sample, fit the circle and count the inliers:
        inl, c, r = ransac_iter()
        score = np.sum(inl)
        # Update the current best:
        if score > score_best:
            score_best, inliers_best, c_best, r_best = score, inl, c, r

    ransac_results.append({'r':r_best, 'c':c_best, 'inliers':data[inliers_best]})
    # Remove all inliers before fitting the next circle
    data = data[np.logical_not(inliers_best)]

def refine_fit(results, method):

    fit_results = []
    for circle in results:

        # Arguments must be a singe vector, therefore concatenate them
        def cost(cr):
            c, r = cr[:2], cr[2]
            if method == 'algebraic':
                return np.sum(np.linalg.norm(circle['inliers']-c, axis=1)**2 - r**2)
            elif method == 'lm':
                return np.sum(np.linalg.norm(circle['inliers']-c, axis=1) - r)

        # Use the RANSAC result as the best guess
        guess = np.concatenate((circle['c'], [circle['r']]))
        fit = least_squares(cost, guess)
        fit_results.append(fit.x)

    return fit_results
            
# Add circles to the diagram using a list of centers and radii
def show_refined_circles(results):

    circle_handles = []
    for method, color in [('algebraic', 'blue'), ('lm', 'orange')]:

        fit_results = refine_fit(results, method)
        for cr in fit_results:
            circle = plt.Circle(cr[:2], radius=cr[2], fill=False, edgecolor=color)
            plt.gca().add_patch(circle)
        # Remember the last circle for the legend
        circle_handles.append(circle)

    plt.legend(handles=circle_handles, labels=['Algebraic distance', 'Levenberg-Marquardt'])

show_cicles_and_data(ransac_results, data)
